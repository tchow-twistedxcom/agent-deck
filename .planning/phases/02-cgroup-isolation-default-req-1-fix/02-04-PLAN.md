```go
// stripSystemdRunPrefix removes the leading "--user --scope --quiet
// --collect --unit <unit> tmux" tokens from args produced by
// startCommandSpec when LaunchInUserScope is true, returning the bare
// tmux args. If the args do not match the expected systemd-run wrap
// shape (e.g. someone changed startCommandSpec without updating this),
// the function returns args unchanged — the caller is then comparing a
// systemd-run argv list against the tmux binary, which will fail
// loudly and predictably.
func stripSystemdRunPrefix(args []string) []string {
    // Expected shape from startCommandSpec:
    //   args[0..5] = "--user", "--scope", "--quiet", "--collect", "--unit", "<unit>"
    //   args[6]    = "tmux"
    //   args[7..]  = original tmux args
    if len(args) >= 7 && args[6] == "tmux" {
        return args[7:]
    }
    return args
}
```

	t.Helper()
	buf := &bytes.Buffer{}

	t.Cleanup(func() { statusLog = original })
	return buf

// failOnLauncher returns an exec wrapper that fails for any command whose
// argv[0] equals failBinary, and otherwise delegates to exec.Command. Used to
// simulate "systemd-run is present but invocation fails" without mutating
// host state.
func failOnLauncher(failBinary string) func(name string, arg ...string) *exec.Cmd {
	return func(name string, arg ...string) *exec.Cmd {
		if name == failBinary {
			// `false` exits 1 — guaranteed failure with no side effects.
			return exec.Command("false")
		}
		return exec.Command(name, arg...)
	}
}
