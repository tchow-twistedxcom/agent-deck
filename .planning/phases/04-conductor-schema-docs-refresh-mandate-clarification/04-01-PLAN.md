// GetConductorClaudeConfigDir returns the conductor-specific Claude config directory,
// if configured. Keyed by conductor name (no "conductor-" prefix). Path expansion
// matches GetGroupClaudeConfigDir. Returns "" when the conductor has no block
// or no config_dir — callers fall through to the group/profile/global chain.
func (c *UserConfig) GetConductorClaudeConfigDir(name string) string {
    if c == nil || name == "" || c.Conductors == nil {
        return ""
